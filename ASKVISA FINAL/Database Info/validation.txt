-- ========== ALSO UPDATE EXISTING DATA FORMAT ==========

-- Update existing date answers to DD-MM-YYYY format
UPDATE applicant_answers aa
JOIN country_questions cq ON aa.question_id = cq.id
SET aa.answer_text = DATE_FORMAT(STR_TO_DATE(aa.answer_text, '%Y-%m-%d'), '%d-%m-%Y')
WHERE cq.field_type = 'date'
  AND cq.country_id = 1
  AND aa.answer_text REGEXP '^[0-9]{4}-[0-9]{2}-[0-9]{2}$';

-- =====================================================
-- ========== VERIFY VALIDATION RULES ===================
-- =====================================================

SELECT
    id,
    field_key,
    label,
    field_type,
    validation_rules,
    JSON_VALID(validation_rules) as is_valid_json
FROM country_questions
WHERE country_id = 1
ORDER BY sort_order;

-- Verify date format updates
SELECT 
    aa.id,
    cq.field_key,
    aa.answer_text as formatted_date,
    STR_TO_DATE(aa.answer_text, '%d-%m-%Y') as parsed_date,
    DATE_FORMAT(STR_TO_DATE(aa.answer_text, '%d-%m-%Y'), '%d-%m-%Y') as reformatted_date
FROM applicant_answers aa
JOIN country_questions cq ON aa.question_id = cq.id
WHERE cq.field_type = 'date'
  AND cq.country_id = 1
LIMIT 10;
